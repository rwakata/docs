---
title: "Get organization"
api: "GET https://api.xente.com/v2/organizations"
description: "This is an object representing your organization. You can use it to retrieve your organizational details."
---

### Query

<ParamField query="id" type="string">
  This is the ID your organization
</ParamField>

### Response

<ResponseField name="statusCode" type="number">
  Indicates whether the call was successful. 0 if successful, error code if not.
</ResponseField>
<ResponseField name="statusMessage" type="string">
  Status message on outcome of the API call
</ResponseField>
<ResponseField name="organization" type="object">

The details of the organization

<Expandable title="Toggle object">

<ResponseField name="id" type="number">
 The unique id of the organization
</ResponseField>
<ResponseField name="logo" type="string">
 The logo of the organization
</ResponseField>
<ResponseField name="name" type="string">
  The name of the organization e.g. My company name
</ResponseField>

<ResponseField name="description" type="string">
  This description of your organization e.g. We sell widgets
</ResponseField>

<ResponseField name="address" type="string">
  This address of the orgarnization
</ResponseField>

<ResponseField name="users" type="JSON or null">
  The user who created the organization
</ResponseField>
<ResponseField name="accounts" type="JSON or null">
  The list of accounts associated with the organization
</ResponseField>
</Expandable>

</ResponseField>
