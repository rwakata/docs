### Query

<ParamField path="personId" type="string">
  Unique id of the person e.g. mobile number, bill_number, passport number, id number etc 
</ParamField>


### Response

<ResponseField name="success" type="number">
  Indicates whether the call was successful. 1 if successful, error code if not.
</ResponseField>

<ResponseField name="data" type="object">

The contents of the person

<Expandable title="Toggle object">

<ResponseField name="personId" type="number">

</ResponseField>


<ResponseField name="name" type="string">
  The name of the person
</ResponseField>

<ResponseField name="image_uri" type="string">
  The image Url of the person
</ResponseField>

<ResponseField name="status" type="string">
  Status of the person 
</ResponseField>

<ResponseField name="person_details" type="JSON or null">
  Other details about the person 
</ResponseField>

</Expandable>

</ResponseField>
