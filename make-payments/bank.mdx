## Step 1 - Verify the person and their association with the bank account.

To ensure your transfering funds to the correct bank account number, please first verify the account name and account number 

<CodeGroup>

```bash cURL

curl -X GET \
     -H "Content-Type: application/json" \
     -H "Authorization: Token <token>" \
     "https://api.xente.co/people/?account_number=<account_number>&account_name=<account_name>&bank=<bank>"

```

```bash Node

const axios = require('axios');

        return null;
      }
    } catch (error) {
      console.log('Error occurred:', error.message);
      return null;
    }
  }
}
```

```bash Java
import okhttp3.*;

        urlBuilder.addQueryParameter("account_name", accountName);
        urlBuilder.addQueryParameter("bank", bank);
        
        String url = urlBuilder.build().toString();
        Request request = new Request.Builder()
                .url(url)
                .build();

                // Handle successful response

                return null;
            }
        } catch (Exception e) {
            // Handle exception

            return null;
        }
    }
}
```

## Step 2 - POST bank transfer transaction

You POST a mobile money transaction as below  

<CodeGroup>

```bash cURL
curl -X POST https://api.xente.co/transactions \
 -H "Content-Type: application/json" \
 -H "Authorization: Token <token>" \
 -d '{
    "clientreferenceid": 67890,
    "batch_id": 54321,
    "account_number": "9876543210",
    "name": "John Doe",
    "currency": "USD",
    "amount": 100.00,
    "transaction_type": "debit",
    "product": {
        "id": "bank_transfer"
    },
    "beneficiary": {
        "id": "256770833993",
        "name": "Halle Berry",
        "data": {
            "bank_account": "999 999 999",
            "bank_name": "DFCU"
        }
    }
}'


```

```bash Java
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public class XenteApi {
    private final OkHttpClient client;
    private final Gson gson;

    public XenteApi() {
        client = new OkHttpClient();
        gson = new Gson();
    }

    public void postTransaction() {
        MediaType mediaType = MediaType.parse("application/json");

        JsonObject product = new JsonObject();
        product.addProperty("id", "bank_transfer");

        JsonObject bank = new JsonObject();
        bank.addProperty("bank_account", "999 999 999");
        bank.addProperty("bank_name", "DFCU");

        JsonObject beneficiary = new JsonObject();
        beneficiary.addProperty("id", "256770833993");
        beneficiary.addProperty("name", "Halle Berry");
        beneficiary.add("data", bank);

        JsonObject transaction = new JsonObject();
        transaction.addProperty("clientreferenceid", 67890);
        transaction.addProperty("batch_id", 54321);
        transaction.addProperty("account_number", "9876543210");
        transaction.addProperty("name", "John Doe");
        transaction.addProperty("currency", "USD");
        transaction.addProperty("amount", 100.00);
        transaction.addProperty("transaction_type", "debit");
        transaction.add("product", product);
        transaction.add("beneficiary", beneficiary);

        RequestBody body = RequestBody.create(mediaType, gson.toJson(transaction));

        Request request = new Request.Builder()
                .url("https://api.xente.co/transactions")
                .addHeader("Content-Type", "application/json")
                .addHeader("Authorization", "Token <token>")
                .post(body)
                .build();

        try {
            Response response = client.newCall(request).execute();

            if (response.isSuccessful()) {
                String responseData = response.body().string();
                JsonObject jsonResponse = JsonParser.parseString(responseData).getAsJsonObject();
                String success = jsonResponse.get("success").toString();
                String data = jsonResponse.get("data").toString();

                String transactionId = JsonParser.parseString(data).getAsJsonObject().get("transaction_id").toString();
                String clientReferenceId = JsonParser.parseString(data).getAsJsonObject().get("clientreferenceid").toString();
                String message = JsonParser.parseString(data).getAsJsonObject().get("message").toString();

                // Handle the response as needed
            } else {
                // Handle the error response
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```

## Step 3 - Receive Instant Payment Notification (IPN)

All transactions on Xente are asynchronous. Therefore, after you POST a mobile money transaction, you need to wait for a success (or failure) response using webhooks called Instant Payment Notifications (IPN).

Learn more about **[Instant Payment Notifications](/start/ipn)**.
