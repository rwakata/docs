import com.google.gson.Gson;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public class XenteApi {
    private final OkHttpClient client;
    private final Gson gson;

    public XenteApi() {
        client = new OkHttpClient();
        gson = new Gson();
    }

    public void postTransaction(String token) throws Exception {
        MediaType mediaType = MediaType.parse("application/json");

        RequestBody body = RequestBody.create(mediaType, "{\n" +
                "  \"id\": \"mtn_mobile_internet\"\n" +
                "}");

        RequestBody beneficiaryBody = RequestBody.create(mediaType, "{\n" +
                "  \"id\": \"256770123456\",\n" +
                "  \"name\": \"Halle Berry\"\n" +
                "}");

        RequestBody transactionBody = RequestBody.create(mediaType, "{\n" +
                "  \"clientreferenceid\": 67890,\n" +
                "  \"batch_id\": 54321,\n" +
                "  \"account_number\": \"9876543210\",\n" +
                "  \"name\": \"John Doe\",\n" +
                "  \"currency\": \"UGX\",\n" +
                "  \"amount\": 10000.00,\n" +
                "  \"transaction_type\": \"debit\",\n" +
                "  \"product\": " + body + ",\n" +
                "  \"beneficiary\": " + beneficiaryBody + "\n" +
                "}");

        Request request = new Request.Builder()
                .url("https://api.xente.co/transactions")
                .post(transactionBody)
                .addHeader("Content-Type", "application/json")
                .addHeader("Authorization", "Token " + token)
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                ResponseData responseData = gson.fromJson(response.body().string(), ResponseData.class);
                boolean success = responseData.isSuccess();
                Object data = responseData.getData();

                // Access the transaction ID and other properties from the response
                String transactionId = data.getTransactionId();
                int clientReferenceId = data.getClientReferenceId();
                String message = data.getMessage();

                // Handle the response as needed
            } else {
                // Handle the error response
            }
        }
    }
}

class ResponseData {
    private boolean success;
    private Object data;

    public boolean isSuccess() {
        return success;
    }

    public Object getData() {
        return data;
    }

| Key                         | Type                   | Description
| --------------------------- | ---------------------- | -------------
| client_reference_id         | string                 | A unique ID from your system to help you identify the transaction
| batch_id                    | string                 | A unique ID to help you identify the batch. This is useful if the transaction is part of a bulk payment
| account_number              | string                 | Your business account number from which funds will be debited
| currency                    | string                 | The currency of your account
| amount                      | number                 | The amount to be debited from your account
| transaction_type               | string                 | Put value "debit" to send mobile money as this debits funds from your account
| product.id                  | string                 | Put the value for Mobile Money Product IDs. You can find IDs under products in the portal
| product.amount              | number                 | The amount of mobile money to be sent
| product.message             | string                 | The message that the beneficiary will receive
| beneficiary.id              | string                 | The unique ID of the beneficiary in your system
| beneficiary.names           | string                 | The names of the beneficiary
| beneficiary.other_information | string                | Other information about the beneficiary





## Step 3 - Receive Instant Payment Notification (IPN)

All transactions on Xente are asynchronous. Therefore, after you POST a mobile money transaction, you need to wait for a success (or failure) response using webhooks called Instant Payment Notifications (IPN).

Learn more about **[Instant Payment Notifications](/start/ipn)**.
