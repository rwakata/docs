With the Xente API, you can easily handle card management tasks right from your own application. This includes simple actions like adding funds, checking card balances, reviewing transaction history, and toggling card activation status.

## Add funds to physical and virtual Visa debit cards

To add funds to individual virtual and physical Visa cards, utilize the following API. It's important to note that the funds will be transferred from your main account balance to the respective individual card account.
<CodeGroup>

```bash cURL

curl -X POST https://api.xente.co/transactions \
 -H "Content-Type: application/json" \
 -H "Authorization: Token <token>" \
 -d '{
    "clientreferenceid": 67890,
    "batch_id": 54321,
    "account_number": "9876543210",
    "name": "John Doe",
    "currency": "UGX",
    "amount": 10000,
    "transaction_type": "debit",
    "product": {
        "id": "visa_debit_card_deposit"
    },
    "beneficiary": {
        "id": "123",
        "name": "Halle Berry",
        "data": {
            "card_id": "1234"
        }
    }
}'



```

```bash Java

import com.google.gson.Gson;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public class XenteApi {
    private final OkHttpClient client;
    private final Gson gson;

    public XenteApi() {
        client = new OkHttpClient();
        gson = new Gson();
    }

    public void postTransaction(String token) throws Exception {
        MediaType mediaType = MediaType.parse("application/json");

        RequestBody body = RequestBody.create(mediaType, "{\n" +
                "  \"id\": \"visa_debit_card_deposit\"\n" +
                "}");

        RequestBody beneficiaryBody = RequestBody.create(mediaType, "{\n" +
                "  \"id\": \"256770123456\",\n" +
                "  \"name\": \"Halle Berry\"\n" +
                "}");

        RequestBody transactionBody = RequestBody.create(mediaType, "{\n" +
                "  \"clientreferenceid\": 67890,\n" +
                "  \"batch_id\": 54321,\n" +
                "  \"account_number\": \"9876543210\",\n" +
                "  \"name\": \"John Doe\",\n" +
                "  \"currency\": \"UGX\",\n" +
                "  \"amount\": 10000.00,\n" +
                "  \"transaction_type\": \"debit\",\n" +
                "  \"product\": " + body + ",\n" +
                "  \"beneficiary\": " + beneficiaryBody + "\n" +
                "}");

        Request request = new Request.Builder()
                .url("https://api.xente.co/transactions")
                .post(transactionBody)
                .addHeader("Content-Type", "application/json")
                .addHeader("Authorization", "Token " + token)
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                ResponseData responseData = gson.fromJson(response.body().string(), ResponseData.class);
                boolean success = responseData.isSuccess();
                Object data = responseData.getData();

                // Access the transaction ID and other properties from the response
                String transactionId = data.getTransactionId();
                int clientReferenceId = data.getClientReferenceId();
                String message = data.getMessage();

                // Handle the response as needed
            } else {
                // Handle the error response
            }
        }
    }
}

class ResponseData {
    private boolean success;
    private Object data;

    public boolean isSuccess() {
        return success;
    }

    public Object getData() {
        return data;
    }
}
```

## Get Visa debit card transaction history

To retrieve card transactions, utilize the following API.

<CodeGroup>

```bash cURL

curl -X GET \
     -H "Content-Type: application/json" \
     -H "Authorization: Token <token>" \
     "https://api.xente.co/cards/?card_id=<card_id>"

```

```bash Node
const axios = require('axios');

const headers = {
  'Content-Type': 'application/json',
  'Authorization': 'Bearer YOUR_AUTH_TOKEN'
};

const data = {
  user_email: 'user@email.com',
  thirdpartyid: '190890'
};

axios.post('https://api.xente.co/cards', data, { headers })
  .then(response => {
    console.log(response.data);
  })
  .catch(error => {
    console.error(error);
  });
```

```bash Java

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;

public class Main {
    public static void main(String[] args) throws Exception {
        String url = "https://api.xente.co/cards";
        String authHeader = "Bearer YOUR_AUTH_TOKEN";
        String requestBody = "{\n" +
                "  \"user_email\": \"user@email.com\",\n" +
                "  \"thirdpartyid\": \"190890\"\n" +
                "}";

        URL obj = new URL(url);
        HttpURLConnection con = (HttpURLConnection) obj.openConnection();

        // Set the request method to POST
        con.setRequestMethod("POST");

        // Set the request headers
        con.setRequestProperty("Content-Type", "application/json");
        con.setRequestProperty("Authorization", authHeader);

        // Enable sending request body
        con.setDoOutput(true);

        // Write the request body
        OutputStream outputStream = con.getOutputStream();
        outputStream.write(requestBody.getBytes());
        outputStream.flush();
        outputStream.close();

        // Get the response
        int responseCode = con.getResponseCode();
        BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
        String inputLine;
        StringBuilder response = new StringBuilder();
        while ((inputLine = in.readLine()) != null) {
            response.append(inputLine);
        }
        in.close();

        // Print the response
        System.out.println("Response Code: " + responseCode);
        System.out.println("Response Body: " + response.toString());
    }
}
```

```bash C#
using System;

                user_email = "user@email.com",

            var content = new StringContent(Newtonsoft.Json.JsonConvert.SerializeObject(requestData));
            content.Headers.ContentType = new MediaTypeHeaderValue("application/json");

            var response = await client.PostAsync(url, content);

            if (response.IsSuccessStatusCode)
            {
                var result = await response.Content.ReadAsStringAsync();
                Console.WriteLine(result);
            }
            else
            {
                Console.WriteLine("Request failed with status code: " + response.StatusCode);
            }
        }
    }
}
```
